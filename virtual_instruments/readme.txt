etd_chords 
  input:  [octave band] [root tone] [intervals] [duration] | <buttons > | [slicing input]
  buttons: 
    green: play
    red: delete last chord
    yellow; play subsection of song (python array slicing)
  bindings:
    h and l to move left and right between input boxes
    enter to play a chord
    r to record a chord
